% ----- Page geometry: 6×9 trade book, asymmetric margins for binding -----
\usepackage[paperwidth=6in, paperheight=9in,
            inner=0.9in, outer=0.65in,
            top=0.85in, bottom=0.85in]{geometry}

% ----- Colors -----
\usepackage[dvipsnames,table]{xcolor}
\colorlet{mmblue}{blue!40!black}
\colorlet{mmintro}{blue!60!black}
\colorlet{mmboxbg}{blue!5!white}

% ----- Empty pages and running headers -----
\usepackage{emptypage}
\usepackage{fancyhdr}
\setlength{\headheight}{15pt}

% Running headers — suppress "Chapter N." prefix in header text
% \pagestyle{fancy} is activated in Main_Book_ManahPrarupe_interior.tex at \mainmatter
% so frontmatter (title, TOC, initialpages) uses LaTeX's default plain style.
\renewcommand{\chaptermark}[1]{\markboth{#1}{#1}}
\fancyhf{}
\fancyhead[LE]{\small\textit{मेन्टल मॉडेल्स}}
\fancyhead[RO]{\small\itshape\leftmark}
\fancyfoot[LE,RO]{\thepage}
\renewcommand{\headrulewidth}{0.4pt}

% Chapter-opening pages: page number only, no running head
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[LE,RO]{\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
}

% ----- Chapter heading design -----
\usepackage{titlesec}
% Format: blue chapter number | title, thin blue rule below
\titleformat{\chapter}[block]
  {\normalfont\bfseries}
  {\textcolor{mmblue}{\normalsize\thechapter\ |\ }}
  {0pt}
  {\LARGE}
  [\vspace{3pt}{\color{mmblue}\titlerule[0.8pt]}]
\titlespacing*{\chapter}{0pt}{0pt}{20pt}

% ----- Hyperlinks -----
\usepackage{hyperref}
\hypersetup{
    colorlinks=true,
    linkcolor=black,
    filecolor=black,
    urlcolor=blue,
    citecolor=black,
    pdfborder={0 0 0}
}

% ----- Math symbols (needed for \blacktriangleright in tcolorbox Try This) -----
\usepackage{amssymb}

% ----- Graphics and floats -----
\usepackage{graphicx}
\usepackage{float}

% ----- tcolorbox with named style -----
% Individual chapter files use inline styles (which match these colors exactly).
% New boxes can use: \begin{tcolorbox}[mmbox, title={N | Topic}]
\usepackage{tcolorbox}
\tcbset{mmbox/.style={
  colback=mmboxbg,
  colframe=mmblue,
  fonttitle=\bfseries\small,
  boxrule=0.5pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt
}}

% ----- Devanagari readability -----
% 10% extra leading — prevents diacritic clashing between lines
\linespread{1.1}

% Looser justification — avoids overfull hboxes in Devanagari word spacing
\tolerance=800
\emergencystretch=2em
\hbadness=10000

% ----- Table of contents heading -----
\renewcommand{\contentsname}{अनुक्रमणिका}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% 'dedication' environment                                                      %
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newenvironment{dedication}
{
   \cleardoublepage
   \thispagestyle{empty}
   \vspace*{\stretch{1}}
   \hfill\begin{minipage}[t]{0.66\textwidth}
   \raggedright
}
{
   \end{minipage}
   \vspace*{\stretch{3}}
   \clearpage
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Chapter quote environment                    %
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\makeatletter
\renewcommand{\@chapapp}{}
\newenvironment{chapquote}[2][2em]
  {\setlength{\@tempdima}{#1}%
   \def\chapquote@author{#2}%
   \parshape 1 \@tempdima \dimexpr\textwidth-2\@tempdima\relax%
   \itshape}
  {\par\normalfont\hfill--\ \chapquote@author\hspace*{\@tempdima}\par\bigskip}
\makeatother
