\section{Simplicity \& Decluttering}

\begin{frame}[fragile]
\frametitle{Ch 1: De-cluttering}
\begin{itemize}
\item If you haven't used it in the last year, junk it
\item Have To-Do lists based on context, like @Home, @Work, @Calls etc
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 2: Single Tasking}
\begin{itemize}
\item Multi-tasking is less efficient
\item Centralize tasks via a single inbox
\item Use Google Tasks + Calendar for scheduling
\item Consolidate email (Outlook, Yahoo, Gmail) into one check-time
\item Keep a diary for short-term notes
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 3: Edit your Commitments}
\begin{itemize}
\item For each area of life there could be goals
\item Just keep 2--3 goals per area
\item Reserve time for relaxation and fun
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 4: Eliminate All but the Essential}
\begin{itemize}
\item Reduce most inputs
\item Cut: TV, frequent websites, newspapers, magazines, RSS feeds, blogs
\item Deactivate useless accounts
\item Do minimum but quality work
\item Be contrarian to stand out
\item Don't spread too thin
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 5: You Will Never Clear Your Task List --- That's OK}
\begin{itemize}
\item A task list is a capture system, not a finish line
\item Focus on your Most Important Tasks (MITs), not completion
\item Accept incompleteness; protect energy for what truly matters
\end{itemize}
\end{frame}

\section{Mindfulness \& Presence}

\begin{frame}[fragile]
\frametitle{Ch 6: Art of Doing Nothing}
\begin{itemize}
\item Close all distractions --- mobile, TV, computer etc.
\item Sit quietly; do attentive breathing
\item While waiting in a queue, resist emails or reading --- just focus on breathing
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 7: Declutter your Mind}
\begin{itemize}
\item Write down everything on your mind, on paper; later route to tasks or a mind map (MM)
\item Office: keep a diary/journal; Personal: keep a small pocket diary for daily notes
\item Do fewer tasks --- do chosen ones well, then relax and enjoy
\item Let go of frustrations, worries, grudges
\item Slow down --- walking, talking, working: no rush
\item Delegate load; set clear expectations
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 8: Simplicity Tips}
\begin{itemize}
\item Remove non-essential commitments
\item Remove non-essential items: books, CDs, electronics, clothes, e-accounts, newsletters, bookmarks
\item Limit communication; close IM; take digital detox breaks
\item Keep work files on work laptop; personal files on home laptop --- no mix-up
\item Read unread books; create mind maps (MMs) of key insights
\item Dedicate Sundays to watching unseen films
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 9: Eat Slower}
\begin{itemize}
\item Don't read or watch anything while eating
\item Embrace the slow (not fast) food movement
\item Pay attention to what you are eating; taste it
\end{itemize}
\end{frame}

\section{Productivity \& Focus}

\begin{frame}[fragile]
\frametitle{Ch 10: Purpose your Day}
\begin{itemize}
\item Identify 3 MITs (Most Important Tasks); tackle the first one early in the morning --- ideally health-related
\item Understand your work deeply; talk to people
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 11: Focus on Big Rocks}
\begin{itemize}
\item Identify weekly tasks that must be accomplished
\item Track them on Google Tasks as well as a whiteboard for visibility
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 12: Clear-out Inbox}
\begin{itemize}
\item Junk $\Rightarrow$ Delete
\item Long / informational $\Rightarrow$ move to appropriate folder as unread
\item Quick reply $\Rightarrow$ respond in 4--5 lines
\item Awaiting response today $\Rightarrow$ keep in Inbox
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 13: Clear your Desk}
\begin{itemize}
\item Keep essential items only
\item Keep work/personal phone PDFs handy
\item Keep ready references, important mind-map PDFs, accounts/password PDF accessible
\item Backup everything on Google Docs and keep a printed set at home and in office
\end{itemize}
\end{frame}

\section{Wellbeing \& Character}

\begin{frame}[fragile]
\frametitle{Ch 14: Become an Early Riser}
\begin{itemize}
\item Rise before the household wakes --- gain quiet, uninterrupted time
\item Use early hours for exercise, planning, or deep work
\item Adjust bedtime gradually to make waking early sustainable
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 15: Stay Motivated in a Slump}
\begin{itemize}
\item Create a goals mind map with pictures and paste it on the wall
\item Consider publicizing goals --- social accountability helps
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 16: Make the Most of your Laziest Day}
\begin{itemize}
\item Relax, take a nap, go for a walk
\item Designate Sunday as a rest day
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 17: Decompress after High Stress}
\begin{itemize}
\item Deep breathing
\item Self massage
\item Take a walk
\item Take a day off --- do nothing at all; yes, anything
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 18: Cultivate Compassion --- Be Kind}
\begin{itemize}
\item Kindness is a practice, not a trait --- it requires deliberate effort
\item Notice opportunities to help; act without waiting to be asked
\item Compassion extends to yourself as well as others
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 19: Escape Materialism}
\begin{itemize}
\item Donate all unnecessary items
\item Keep minimum --- but quality --- possessions
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 20: Practical Tips for Living with a Golden Rule}
\begin{itemize}
\item Treat others the way you want to be treated
\item Be helpful, kind, and friendly
\item Stop criticism
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 21: Accept Criticism with Grace and Appreciation}
\begin{itemize}
\item Try to explain your stance to give clarity
\item ``I don't think I made myself very clear\ldots''
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 22: Have Faith in Humanity and Restore Kindness}
\begin{itemize}
\item People are meeting less --- becoming ruder and lonelier
\item Practice kindness; help others without expecting anything in return
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 23: Boost your Self Confidence}
\begin{itemize}
\item Groom yourself: good language, good looks, good dressing
\item Know yourself: observe predominant thoughts; identify faults, limitations, fears
\item Speak slowly; stand tall and smile
\item Increase competence; volunteer
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Ch 24: Live your Life Consciously}
\begin{itemize}
\item Have clarity on what you are doing and why
\item Take control of your life; consider your impact on the world
\item Write the best books --- perhaps free e-books
\item Write the best apps and release them free or at nominal charge
\item Give project/thesis guidance for free
\item Build leaders and teams
\end{itemize}
\end{frame}
