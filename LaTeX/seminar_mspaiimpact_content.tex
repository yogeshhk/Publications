\end{block}

\begin{block}{कविता B}
पाऊस आला, आठवणी जाग्या झाल्या  \\

ओल्या मातीमध्ये बालपण सापडलं  \\

छपरावरच्या थेंबांत गाणं ऐकलं  \\

मन नकळत पुन्हा हरवलं
\end{block}

\vspace{0.25cm}
{\small एक कविता AI ने तयार केली आहे. एक माणसाने लिहिलेली आहे.}\\
\textbf{तुम्ही ओळखू शकता का?}

\note{Reveal after audience guesses: Poem A is AI-generated (GPT-4); Poem B is human-written.
      The difficulty of telling them apart is the central tension of this seminar.

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{हा विषय आज का महत्त्वाचा?}

\begin{itemize}
\item AI वेगाने दैनंदिन जीवनाचा, लेखनाचा, शिक्षणाचा भाग बनत आहे
\item \textbf{२०२४:} ChatGPT वापरकर्ते जगभरात $> 18$ कोटी; भारतात $> 2$ कोटी
\item मराठी भाषेसाठी मोठी संधी \textit{आणि} तितकाच धोका
\item साहित्यिक, शिक्षक, विद्यार्थी — प्रत्येकजण आज निर्णयाच्या उंबरठ्यावर
\end{itemize}

\vspace{0.2cm}
\begin{alertblock}{मुख्य प्रश्न}
AI मराठी भाषा आणि साहित्याला \textbf{समृद्ध} करेल की \textbf{दुर्बल}?
\end{alertblock}

\note{Frame the whole talk around this one central question.
      Every subsequent slide is evidence for or against one side.}

\end{frame}

%% SLIDE 3  |  AI म्हणजे काय आणि ते कसे शिकते?
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{AI म्हणजे काय आणि ते कसे शिकते?}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{block}{AI = मानवी बुद्धिमत्तेचे अनुकरण}
\begin{itemize}
\item बुद्धिबळ खेळणे, भाषा समजणे, चित्र ओळखणे
\item \textbf{परीक्षेची उपमा:} काही जण संकल्पना समजून घेतात; काही जण फक्त जुने प्रश्नपत्रिका सोडवतात
\item AI हा \textit{दुसऱ्या} प्रकारचा विद्यार्थी — जुन्या data मधून pattern शिकतो
\item \textbf{जास्त data = अधिक चांगला AI}
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Generative AI म्हणजे काय?}
\begin{itemize}
\item उत्तर \textit{देत नाही} — उत्तर \textbf{तयार करतो}
\item \textbf{SMS उपमा:} तुमचे मागचे १०० messages वाचून पुढचा शब्द सुचवतो — ``चला जाऊया \ldots'' $\rightarrow$ \textit{coffee} की \textit{सिनेमा}?
\item ChatGPT हे याच तत्त्वावर, फक्त अब्जावधी वाक्यांवर प्रशिक्षित
\end{itemize}
\vspace{0.1cm}
\textbf{AI = शिष्य \quad|\quad Prompt = गुरु}
\end{block}
\end{column}
\end{columns}

\vspace{0.2cm}
\begin{alertblock}{लक्षात ठेवा}
AI चे उत्तर ``ज्ञान'' नव्हे, \textit{अंदाज} असतो — म्हणून स्वतःचा विचार अत्यावश्यक.
\end{alertblock}

\note{Exam analogy works very well with this audience — pause and ask them which type
      of student they were. Then connect: ML is the rote-learning student who has read
      every past paper ever written, but has no real understanding.
      SMS analogy: most people have seen T9 / SwiftKey suggestions — scale that up
      to the entire internet and you have GPT.}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% SLIDE 4  |  AI ला इतका Data कुठून मिळाला?
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{AI ला इतका Data कुठून मिळाला?}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{block}{इंटरनेटने शक्य केले}
\begin{itemize}
\item आपले messages, emails, Facebook posts
\item डिजिटल वृत्तपत्रे, विकिपीडिया, ब्लॉग्ज
\item डिजिटायझेशन केलेली पुस्तके व हस्तलिखिते
\item YouTube चे subtitles, tweets, forums
\item \textbf{२०२३ पर्यंत:} $> 1$ ट्रिलियन वेबपेजेस उपलब्ध
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Data आणि गुणवत्ता}
\begin{itemize}
\item जास्त data $\Rightarrow$ AI अधिक सक्षम होतो
\item इंग्रजीत अफाट data $\Rightarrow$ ChatGPT इंग्रजीत उत्कृष्ट
\item मराठीत data कमी $\Rightarrow$ \textbf{मराठी AI अजून कमकुवत}
\item आपले मराठी लेखन, आपला मराठी data
\end{itemize}
\end{block}
\end{column}
\end{columns}

\vspace{0.2cm}
\begin{exampleblock}{आपण काय करू शकतो?}
मराठी Wikipedia संपादन $\cdot$ मराठीत ब्लॉग लिहिणे $\cdot$ Bhashini साठी voice data देणे
— \textbf{हे सर्व AI ला मराठी शिकवणे आहे!}
\end{exampleblock}

\note{This slide makes the audience feel empowered — their Marathi writing is literally
      training data. Every Marathi Wikipedia article they write improves future AI tools.
      The data gap between English (~46% of web) and Marathi (~0.07%) is stark and worth
      mentioning to motivate action.}

\end{frame}

\begin{columns}[T]

\begin{itemize}

\item मराठी $\leftrightarrow$ इंग्रजी भाषांतर
\item कथा, भाषण, निबंधाचा मसुदा
\item मोठ्या मजकुराचा त्वरित सारांश
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{शिक्षण व व्यवसाय}
\begin{itemize}
\item विद्यार्थ्यांना वैयक्तिक मार्गदर्शन
\item शिक्षकांसाठी प्रश्नपत्रिका व नोट्स
\item YouTube स्क्रिप्ट, सोशल मीडिया पोस्ट
\item पत्रकारांचे बातम्यांचे ड्राफ्ट
\end{itemize}
\end{block}
\end{column}
\end{columns}

\vspace{0.2cm}
\begin{exampleblock}{२०२५ मधील प्रमुख साधने}
\textbf{Google Gemini} मराठीत थेट संवाद $\cdot$
\textbf{AI Anuvaad} (C-DAC) भारतीय भाषांतर $\cdot$
\textbf{Sarvam-1} भारतीय भाषांसाठी open-source LLM $\cdot$
\textbf{Bhashini} २२ भारतीय भाषा
\end{exampleblock}

\note{Sarvam-1 (2024) is India's first open-source LLM trained on Indian languages including
      Marathi. AI Anuvaad is C-DAC's official translation engine.
      Demo opportunity: if internet available, show a live Bhashini or Gemini Marathi query.}

\end{frame}

%% SLIDE 6  |  धोके, आव्हाने आणि नोकऱ्यांवर परिणाम
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{धोके, आव्हाने आणि नोकऱ्यांवर परिणाम}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{alertblock}{AI चे धोके}
\begin{itemize}
\item \textbf{Hallucination:} AI आत्मविश्वासाने चुकीची माहिती देतो — काही माणसांसारखेच!
\item मौलिकता कमी — सर्व लेखन एकसारखे
\item सांस्कृतिक संदर्भ चुकीचे (मराठी सण, परंपरा)
\item बोलीभाषांचा व मराठी अस्मितेचा ऱ्हास
\end{itemize}
\end{alertblock}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{नोकऱ्यांवर परिणाम (WEF 2025)}
\begin{itemize}
\item \textbf{धोका:} साध्या, पुनरावृत्तीच्या कामांना — data entry, साधे भाषांतर, basic content
\item \textbf{सध्या सुरक्षित:} शारीरिक कौशल्याची कामे — robot अजून पूर्ण सक्षम नाही
\item \textbf{नवीन संधी:} AI training, prompt engineering, content review
\end{itemize}
\end{block}
\end{column}
\end{columns}

\vspace{0.2cm}
\begin{exampleblock}{विद्यार्थ्यांची स्वतंत्र विचारशक्ती}
AI वर अवलंबून राहिल्यास भाषिक क्षमता क्षीण होते — लेखन हे \textit{विचाराचे} साधन आहे, फक्त उत्पादनाचे नव्हे.
\end{exampleblock}

\note{Hallucination: AI confidently states false facts — classic example is AI inventing
      fake court cases (the US lawyer incident, 2023). Relate to Marathi: ask AI about a
      local saint or historical figure and it may confabulate dates and events.
      Physical work safety: robots still struggle with unstructured environments —
      a plumber, carpenter or farmer is safer than a content writer right now.}

\end{frame}

\begin{frame}[fragile]\frametitle{कॉपीराइट, नैतिकता आणि खुले प्रश्न}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{alertblock}{नैतिक प्रश्न}
\begin{itemize}
\item AI-generated मजकुराचा मालक कोण?
\item लेखकांच्या संमतीशिवाय AI प्रशिक्षण?
\item साहित्यचौर्य व बनावट श्रेयाचा धोका
\item \textbf{AI content उघड करण्याची जबाबदारी}
\end{itemize}
\end{alertblock}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{विचार करण्यासारखे}
\begin{itemize}
\item भावना नसताना लिहिलेली कविता खरी का?
\item AI साहित्याला पुरस्कार द्यावेत का?
\item भविष्यात लेखकाची भूमिका काय?
\item तंत्रज्ञान आणि संस्कृतीचा समतोल कसा?
\end{itemize}
\end{block}
\end{column}
\end{columns}

\vspace{0.2cm}
\begin{exampleblock}{जागतिक घडामोडी (२०२४--२५)}
न्यूयॉर्क टाइम्स वि. OpenAI खटला $\cdot$ EU AI Act लागू $\cdot$ भारत: AI नियामक धोरण प्रतीक्षेत
\end{exampleblock}

\note{NYT lawsuit (Dec 2023): OpenAI trained on millions of NYT articles without permission
      — directly analogous to concerns about Marathi literary works being scraped.
      EU AI Act (Aug 2024) mandates disclosure of AI-generated content.
      India's Digital India Act is in draft and expected to address AI specifically.}

\end{frame}

\begin{frame}[fragile]\frametitle{धोरण, भविष्य आणि आपली भूमिका}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{block}{AI थांबवता येणार नाही — तर काय करावे?}
\begin{itemize}
\item \textbf{Prompt Engineering} शिका — AI कडून चांगले काम करवून घ्यायचे कौशल्य
\item ChatGPT, Gemini, Copilot — साधने वापरायला शिका
\item \textbf{आपल्या क्षेत्रात ChatGPT पेक्षा चांगले व्हा!}
\item साध्या लेखनात AI जिंकेल; \textit{मराठी तांत्रिक, सांस्कृतिक, भावनिक लेखनात} तुम्ही जिंकू शकता
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{धोरणात्मक गरजा}
\begin{itemize}
\item मराठी AI साधनांना सरकारी निधी
\item हस्तलिखिते व ग्रंथालयांचे डिजिटायझेशन
\item शालेय अभ्यासक्रमात AI साक्षरता
\item लेखकांच्या बौद्धिक संपदा हक्कांचे कायदे
\end{itemize}
\end{block}
\end{column}
\end{columns}

\vspace{0.2cm}
\begin{exampleblock}{तुम्ही आत्ता काय करू शकता?}
मराठी Wikipedia संपादन $\cdot$ Bhashini app वापरून अभिप्राय द्या $\cdot$
Prompt Engineering मोफत शिका (YouTube, NPTEL) $\cdot$ AI content वाचताना स्रोत तपासा
\end{exampleblock}

\note{Key motivational message: for general population writing (social media posts, generic
      articles) AI will be competitive. But for writing a Marathi technical paper, a poem
      about a specific village, or a nuanced cultural commentary — a knowledgeable human
      still wins. The goal is to find that niche and be irreplaceably good at it.
      Prompt engineering courses: Andrew Ng's free course on Coursera, NPTEL AI modules.}

\end{frame}
